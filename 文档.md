成熟下载器系统设计文档 (Rust)
1. 设计哲学与核心原则
1.1 三大目标平衡
成功率 (Success Rate)
    ↑
    |    效率 (Efficiency)
    |      ↑
    |     / \
    |    /   \
    |   /     \
    +----------------→ 安全性 (Safety)
核心原则：
单一事实来源：任何时候只有一个下载任务在执行
故障快速转移：工具失败时立即切换，不阻塞
资源硬隔离：每个下载工具运行在独立沙箱中
可观测性：所有决策都有日志和指标

    /// 优先级 (数字越小优先级越高)
    fn priority(&self) -> u8;
}
3.2 任务调度器 (Scheduler)
pub struct DownloadScheduler {
    /// 后端注册表
    backends: HashMap<BackendId, Box<dyn DownloadBackend>>,
    
    /// 并发控制信号量
    semaphore: Arc<Semaphore>,
    
    /// 任务队列
    task_queue: Arc<Mutex<VecDeque<DownloadTask>>>,
    
    /// 活跃任务
    active_tasks: Arc<DashMap<TaskId, ActiveTask>>,
    
    /// 配置
    config: SchedulerConfig,
}

impl DownloadScheduler {
    /// 核心调度逻辑
    pub async fn schedule(&self, task: DownloadTask) -> Result<DownloadHandle> {
        // 1. 验证任务
        self.validate_task(&task)?;
        
        // 2. 获取并发许可
        let permit = self.semaphore.acquire().await?;
        
        // 3. 探测可用后端
        let candidates = self.probe_backends(&task).await;
        
        // 4. 选择最优后端
        let selected = self.select_best_backend(candidates).await?;
        
        // 5. 启动下载
        self.execute_download(selected, task, permit).await
    }
}
3.3 测速系统 (Probe System)
pub struct ProbeSystem {
    /// 测速策略
    strategy: ProbeStrategy,
    
    /// 历史性能数据
    performance_history: Arc<RwLock<PerformanceHistory>>,
    
    /// 超时配置
    timeout: Duration,
}

impl ProbeSystem {
    /// 执行测速
    pub async fn probe_url(&self, url: &Url) -> Result<ProbeReport> {
        let backends = self.get_available_backends().await?;
        
        // 并行测速，但限制并发
        let results = futures::stream::iter(backends)
            .map(|backend| self.probe_single_backend(backend, url))
            .buffer_unordered(self.config.max_concurrent_probes)
            .collect::<Vec<_>>()
            .await;
        
        Ok(ProbeReport::from_results(results))
    }
    
    /// 单个后端测速
    async fn probe_single_backend(
        &self,
        backend: &dyn DownloadBackend,
        url: &Url,
    ) -> ProbeResult {
        // 1. HEAD 请求检查可访问性
        let head_result = self.quick_head_check(url).await;
        
        // 2. 小样本下载测速
        let speed_test = self.sample_download(url, 256 * 1024).await;
        
        // 3. 稳定性评估
        let stability = self.evaluate_stability(&speed_test);
        
        ProbeResult {
            backend_id: backend.id(),
            latency: head_result.latency,
            throughput: speed_test.avg_speed,
            stability_score: stability.score,
            success_rate: stability.success_rate,
        }
    }
}

pub struct SandboxedExecutor {
    /// 资源限制
    resource_limits: ResourceLimits,

    /// 文件系统隔离

impl SandboxedExecutor {
    pub async fn execute_command(
        &self,
        command: Command,
    ) -> Result<ExecutionResult> {
        // 1. 设置进程资源限制
        let limits = self.apply_resource_limits();
        
        // 2. 配置网络命名空间
        let netns = self.setup_network_namespace();
        
        // 3. 设置文件系统挂载点
        let mount_ns = self.setup_filesystem_namespace();
        
        // 4. 执行命令
        self.run_in_sandbox(command, limits, netns, mount_ns).await
    }
}
4.2 输入验证
pub struct InputValidator {
    /// URL 白名单
    allowed_domains: Vec<DomainPattern>,
    
    /// 协议限制
    allowed_protocols: Vec<Protocol>,
    
    /// 路径遍历防护
    path_validator: PathValidator,
}

impl InputValidator {
    pub fn validate_download_request(
        &self,
        request: &DownloadRequest,
    ) -> Result<ValidatedRequest> {
        // 1. 验证 URL 格式
        let url = Url::parse(&request.url)?;
        
        // 2. 检查协议
        self.validate_protocol(&url)?;
        
        // 3. 检查域名
        self.validate_domain(&url)?;
        
        // 4. 验证文件路径
        let safe_path = self.path_validator.validate(&request.target_path)?;
        
        Ok(ValidatedRequest { url, safe_path })
    }
}

5.2 智能重试策略
pub struct RetryStrategy {
    /// 最大重试次数
    max_retries: u32,
    
    /// 退避策略
    backoff: BackoffPolicy,
    
    /// 错误分类器
    error_classifier: ErrorClassifier,
}

impl RetryStrategy {
    pub async fn execute_with_retry<F, T>(
        &self,
        operation: F,
    ) -> Result<T>
    where
        F: Fn() -> Pin<Box<dyn Future<Output = Result<T>>>>,
    {
        let mut attempt = 0;
        let mut delay = Duration::from_millis(100);
        
        loop {
            match operation().await {
                Ok(result) => return Ok(result),
                Err(err) => {
                    attempt += 1;
                    
                    if !self.should_retry(&err, attempt) {
                        return Err(err);
                    }
                    
                    // 指数退避 + 抖动
                    delay = self.calculate_delay(attempt);
                    tokio::time::sleep(delay).await;
                }
            }
        }
    }
}

pub struct MetricsCollector {
    registry: Registry,
    
    /// 下载计数器
    downloads_total: Counter,
    
    /// 下载耗时直方图
    download_duration: Histogram,
    
    /// 后端成功率
    backend_success_rate: Gauge,
    
    /// 当前活跃下载数
    active_downloads: Gauge,
}

impl MetricsCollector {
    pub fn record_download_start(&self, backend: &BackendId) {
        self.downloads_total
            .with_label_values(&[backend.as_str()])
            .inc();
    }
    
    pub fn record_download_complete(
        &self,
        backend: &BackendId,
        duration: Duration,
        size: u64,
    ) {
        self.download_duration
            .with_label_values(&[backend.as_str()])
            .observe(duration.as_secs_f64());
    }
}

9. 最佳实践总结
✅ DO
使用 tokio::sync::Semaphore控制并发
为每个后端实现健康检查
记录详细的指标和日志
实现优雅的降级策略
定期清理临时文件
❌ DON'T
不要在外部工具中嵌套创建线程
不要信任外部程序的输出
不要无限重试同一个后端
不要忽略资源限制
不要在主线程中执行阻塞操作
10. 最终建议
成熟的下载器不是功能最多，而是：
在最坏情况下仍然可控
在任何环境下都能完成基本下载
永远不会消耗超过预期的系统资源
所有行为都可预测、可观测、可回滚
这个设计可以保证你在 Rust + 多工具混合​ 的情况下，既享受高性能，又避免线程爆炸和安全风险。
